def fills(order_id, params = {})
  operational_url = "#{PORTFOLIO_PATH}/orders/#{order_id}/fills"
  query_params = build_query_params(params)
  operational_url += '/?' + query_params unless query_params.empty?

  format_response(get(operational_url))
end
